```Excel
=LET(
    源数据,
    区域算表!$A$2 # ,
    全标题,

    部门列,
    CHOOSECOLS(源数据, 2),

    判断矩阵,

    CHOOSECOLS(源数据, SEQUENCE(1, 9, 5)),

    MMULT(判断矩阵, 可加数据),
    本月净额,
    CHOOSECOLS(聚合数值, 2) + CHOOSECOLS(聚合数值, 3),
    目标进度,
    IFERROR(本月净额 / CHOOSECOLS(聚合数值, 1), "-"),
    同比进度,
    IFERROR(本月净额 / (CHOOSECOLS(聚合数值, 4) + CHOOSECOLS(聚合数值, 5)), "-"),
    同比差值,
    本月净额 - (CHOOSECOLS(聚合数值, 4) + CHOOSECOLS(聚合数值, 5)),
    目标差值,
    本月净额 - CHOOSECOLS(聚合数值, 1),
    全数据,
    HSTACK(聚合数值, 同比进度, 目标进度, 同比差值, 目标差值),
    列头,
    CHOOSECOLS(全标题, SEQUENCE(1, 13, 5)),
    上部分,
    HSTACK("部门", 列头),
    下部分,
    HSTACK(行头, 全数据),
    VSTACK(上部分, 下部分)
)
```
