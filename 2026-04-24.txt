这次跟着昨天那个函数的计算步骤一步一步拆出来MMULT条件求和是怎么运作的，虽然不熟但是已经完全弄明白了：
把条件列转置之后和行数为m的特征列做一个相等计算，得到一个m*n的布尔矩阵，再把这个布尔矩阵转置成一个n*m的判断矩阵，再用MMULT去和m行的数据列做一个乘积就搞定了。
这次是正儿八经自己写的。
```Excel
=LET(
    名单,
    UNIQUE(CHOOSECOLS(商品算表!A2 # , 5)),
    条件求和,
    LAMBDA(
        a,
        LET(
            data,
            CHOOSECOLS(商品算表!A2 # , MATCH(a, 商品算表!$A$1 # , 0)),
            特征列,
            CHOOSECOLS(商品算表!A2 # , 5),
            判断矩阵,
            TRANSPOSE( --(TRANSPOSE(名单) = 特征列)),
            MMULT(判断矩阵, data)
        )
    ),
    本月销售额,
    条件求和("本月销售额"),
    本月退货额,
    条件求和("本月退货额"),
    去年销售额,
    条件求和("去年同期销售额"),
    去年退货额,
    条件求和("去年同期退货额"),
    上月销售额,
    条件求和("上月销售额"),
    上月退货额,
    条件求和("上月退货额"),
    净销售额,
    本月销售额 + 本月退货额,
    去年净销售额,
    去年销售额 + 去年退货额,
    上月净销售额,
    上月销售额 + 上月退货额,
    同比变化,
    IFERROR((净销售额 / 去年净销售额) - 1, 0),
    同比变动,
    净销售额 - 去年净销售额,
    环比变化,
    IFERROR((净销售额 / 上月净销售额) - 1, 0),
    环比变动,
    净销售额 - 上月净销售额,
    输出,
    HSTACK(名单, 本月销售额, 本月退货额, 净销售额, 去年净销售额, 上月净销售额, 同比变化, 同比变动, 环比变化, 环比变动),
    SORT(输出, MATCH($B$3, $B$5 # , 0), IF($C$3 = "升序", 1, - 1))
)
```
